# 元组参数 def func_arg(*args):     print("args:", args) func_arg(     1,     "youzan",     'dba',     '你好' ) # 字典参数 def func_kwargs(**kwargs):     print("kwargs:", kwargs) func_kwargs(     id=1,     name='youzan',     city='hangzhou',     age='20',     txt='来日方长的' )

args: (1, ‘youzan’, ‘dba’, ‘你好’)
kwargs: {‘id’: 1, ‘name’: ‘youzan’, ‘city’: ‘hangzhou’, ‘age’: ‘20’, ‘txt’: ‘来日方长的’}
